\section{Methodology}
\label{sec:method}

This section presents the full mathematical development of \textsc{Siren}.
We begin with notation and a formal problem statement, then develop the
graph construction procedure, the graph-coupled It\^o SDE model and its
stationary distribution, the score network and its training objective, and
finally the Stein-residual inference algorithm together with four theoretical
guarantees.
The overall pipeline is illustrated in Figure~\ref{fig:arch_diagram}.

\begin{figure}[htpb]
    \centering
    \includegraphics[width=\linewidth]{figures/arch_diagram.eps}
    \caption{%
        The \textsc{Siren} pipeline.  Raw traffic logs are aggregated into
        fixed windows to build dynamic flow graphs.  The joint node-state
        trajectory is modelled by a graph-coupled It\^o SDE whose stationary
        distribution $\pi^{*}$ is characterised by a learned score network.
        At inference, deviations from $\pi^{*}$ are measured node-wise via
        the Stein residual and propagated over the graph to surface
        coordinated intrusions.
    }
    \label{fig:arch_diagram}
\end{figure}

% ---------------------------------------------------------
\subsection{Notation and Problem Setup}
\label{sec:notation}
% ---------------------------------------------------------

Let a monitored network at time $t$ be represented by a weighted graph
$\mathcal{G} = (V, E, A)$, where $V$ is the set of $n = |V|$ active host
addresses, $E \subseteq V \times V$ is the set of observed connections, and
$A \in \mathbb{R}^{n \times n}_{\ge 0}$ is the weighted adjacency matrix.
The graph Laplacian is $L = D - A$, where $D = \operatorname{diag}(A\mathbf{1})$.
Each node $i \in V$ carries a traffic-feature state $X_i(t) \in \mathbb{R}^d$;
the joint state is $X(t) \in \mathbb{R}^{n \times d}$.

We adopt the following regularity assumptions, which underpin all subsequent
theoretical results.

\begin{assumption}[Lipschitz drift]\label{ass:lip}
$f_\theta : \mathbb{R}^d \to \mathbb{R}^d$ is globally Lipschitz:
$\|f_\theta(x) - f_\theta(y)\| \le L_f \|x - y\|$ for all $x, y \in \mathbb{R}^d$.
\end{assumption}

\begin{assumption}[Non-degenerate diffusion]\label{ass:nondeg}
$\sigma_\phi : \mathbb{R}^d \to \mathbb{R}^{d \times d}$ satisfies
$\sigma_\phi(x)\sigma_\phi(x)^{\!\top} \ge \lambda_{\min} I_d$ uniformly in
$x$, for some constant $\lambda_{\min} > 0$.
\end{assumption}

\begin{assumption}[Dissipativity]\label{ass:diss}
There exist constants $\alpha > 0$ and $\beta \ge 0$ such that
$\langle x,\, f_\theta(x) \rangle \le -\alpha\|x\|^2 + \beta$
for all $x \in \mathbb{R}^d$.
\end{assumption}

\textbf{Formal objective.}  We model \emph{benign} traffic as a realisation
of a stochastic process that, under Assumptions~\ref{ass:lip}--\ref{ass:diss},
converges to a unique stationary distribution $\pi^* \in \mathcal{P}(\mathbb{R}^d)$
for each node.  Let $s^*(x) = \nabla_x \log \pi^*(x)$ be the
\emph{score function} of $\pi^*$.  We approximate $s^*$ by a neural network
$s_{\theta\phi}$ trained on benign data.  At inference time we declare node
$i$ anomalous at time $t$ if its empirical trajectory $X_i(t)$ violates the
Stein identity $\mathbb{E}_{\pi^*}[\mathcal{A}_{\pi^*} g] = 0$, as quantified
by the \emph{Stein residual} $\tilde{R}_i(t)$ defined in
Section~\ref{sec:inference}.

% ---------------------------------------------------------
\subsection{Graph Construction and Feature Extraction}
\label{sec:graph}
% ---------------------------------------------------------

\textsc{Siren} processes a stream of flow records in sliding windows of size
$\Delta$ seconds.

\textbf{Adjacency.}  Within each window, we draw a directed edge from host
$i$ to host $j$ if at least $k_{\min}$ flows are observed.  Edge weights are
log-scaled to dampen the influence of traffic volume spikes:
\begin{equation}\label{eq:weight}
A_{ij}
= \frac{\log\!\bigl(1 + \mathrm{flows}(i\to j)\bigr)}
       {\log\!\bigl(1 + \mathrm{flows\_out}(i)\bigr)}.
\end{equation}
For undirected analysis we symmetrise $A \leftarrow (A + A^\top)/2$.
The Laplacian $L$ is recomputed after each window update.

\textbf{Node features.}
The $d\!=\!12$ dimensional feature vector $X_i(t) \in \mathbb{R}^d$ is:
\begin{equation}\label{eq:feat}
X_i(t) = \bigl[
    \bar{\tau},\;
    \log(1{+}\mathcal{B}_{\mathrm{tx}}),\;
    \log(1{+}\mathcal{B}_{\mathrm{rx}}),\;
    \log(1{+}N_p),\;
    \bar{\delta},\;
    \hat{\sigma}_\delta,\;
    |\mathcal{P}_{\mathrm{dst}}|,\;
    |\mathcal{P}_{\mathrm{src}}|,\;
    \rho_{\mathrm{tcp}},\;
    \rho_{\mathrm{udp}},\;
    \rho_{\mathrm{syn}},\;
    \textstyle\sum_j A_{ij}
\bigr]^\top,
\end{equation}
where $\bar{\tau}$ is mean flow duration, $\mathcal{B}$ bytes sent/received,
$N_p$ packet count, $\bar{\delta}$ and $\hat{\sigma}_\delta$ the mean and
standard deviation of inter-arrival times, $\mathcal{P}$ the sets of unique
ports, $\rho$ the protocol and SYN-rate ratios, and $\sum_j A_{ij}$ the
weighted node degree.
All features are $Z$-score normalised using $(\mu_\mathrm{tr},
\sigma_\mathrm{tr})$ estimated on the benign training split only.

% ---------------------------------------------------------
\subsection{Graph-Coupled It\=o SDE Model}
\label{sec:sde}
% ---------------------------------------------------------

\subsubsection{Continuous-Time Dynamics}

For each node $i \in V$, we postulate the following It\^o SDE:
\begin{equation}\label{eq:sde}
dX_i(t) = \underbrace{f_\theta(X_i(t))}_{\text{autonomous drift}}
         + \underbrace{\gamma \sum_{j \in \mathcal{N}(i)}
           A_{ij}\bigl(X_j(t) - X_i(t)\bigr)}_{\text{graph Laplacian coupling}}\; dt
         + \underbrace{\sigma_\phi(X_i(t))\, dW_i(t)}_{\text{stochastic diffusion}},
\end{equation}
where $W_i(t)$ is a $d$-dimensional standard Wiener process and
$\gamma \ge 0$ is the coupling strength.
Writing the $nd$-dimensional joint state
$\mathbf{X}(t) = \operatorname{vec}(X(t))$ and
$\mathbf{F}_\theta(\mathbf{X}) = \operatorname{vec}(F_\theta(X))$, the system
takes the compact matrix form:
\begin{equation}\label{eq:sde_matrix}
d\mathbf{X}(t)
= \Bigl[\mathbf{F}_\theta(\mathbf{X}(t)) - \gamma(L \otimes I_d)\,\mathbf{X}(t)\Bigr]\,dt
+ \boldsymbol{\Sigma}_\phi(\mathbf{X}(t))\,d\mathbf{W}(t),
\end{equation}
where $\otimes$ is the Kronecker product and
$\boldsymbol{\Sigma}_\phi = \mathrm{diag}(\sigma_\phi(X_1),\ldots,\sigma_\phi(X_n))$.
The Laplacian coupling $-(L\otimes I_d)\mathbf{X}$ is negative semi-definite
in $\mathbf{X}$ and therefore adds a mean-reverting tendency: nodes with
similar traffic profiles attract each other in feature space, encoding a
graph-structural prior on normal behaviour.
The diffusion $\sigma_\phi$ is parameterised as the Cholesky factor of a
positive-definite matrix, ensuring $\boldsymbol{\Sigma}_\phi\boldsymbol{\Sigma}_\phi^\top
\succ 0$ everywhere.

\subsubsection{Fokker--Planck Equation and Stationary Distribution}

The marginal density $p_t^{(i)}(x)$ of node $i$'s state satisfies the
Fokker--Planck (forward Kolmogorov) equation:
\begin{equation}\label{eq:fp}
\frac{\partial p_t^{(i)}}{\partial t}
= -\nabla_x \cdot \Bigl[b_i^{(t)}(x)\, p_t^{(i)}(x)\Bigr]
  + \frac{1}{2}\sum_{k,l}\frac{\partial^2}{\partial x_k \partial x_l}
    \Bigl[\bigl(\sigma_\phi \sigma_\phi^\top\bigr)_{kl}(x)\, p_t^{(i)}(x)\Bigr],
\end{equation}
where $b_i^{(t)}(x)$ is the effective drift of node $i$ at time $t$.
The stationary distribution $\pi^*$, if it exists, satisfies
$\partial_t p_t^{(i)} = 0$, i.e.\ the right-hand side of
\eqref{eq:fp} equals zero.
We now establish existence and uniqueness under
Assumptions~\ref{ass:lip}--\ref{ass:diss}.

\begin{theorem}[Ergodicity and unique stationary distribution]
\label{thm:ergodic}
Under Assumptions~\ref{ass:lip}, \ref{ass:nondeg}, and \ref{ass:diss},
the graph-coupled It\^o SDE~\eqref{eq:sde} is geometrically ergodic: there
exists a unique invariant probability measure $\pi^* \in
\mathcal{P}(\mathbb{R}^d)$ such that for every initial distribution
$\mu_0 \in \mathcal{P}(\mathbb{R}^d)$ with finite second moment,
\begin{equation}
W_2(\mu_0 P_t,\, \pi^*) \le C_0\, e^{-\rho t},
\quad t \ge 0,
\end{equation}
where $P_t$ is the Markov semigroup of \eqref{eq:sde},
$W_2$ is the $2$-Wasserstein distance,
and $C_0, \rho > 0$ depend on $L_f, \lambda_{\min}, \alpha, \beta, \gamma$.
\end{theorem}

\begin{proof}[Proof sketch]
Define the Lyapunov function $V(\mathbf{X}) = \|\mathbf{X}\|^2 + c$ for a
constant $c > 0$.
Applying It\^o's formula to $V$ along trajectories of~\eqref{eq:sde_matrix}:
\begin{align}
\mathcal{L} V(\mathbf{X})
&= 2\mathbf{X}^\top\Bigl[\mathbf{F}_\theta(\mathbf{X})
   - \gamma(L\otimes I_d)\mathbf{X}\Bigr]
   + \operatorname{tr}\bigl(\boldsymbol{\Sigma}_\phi\boldsymbol{\Sigma}_\phi^\top\bigr),
\label{eq:genV}
\end{align}
where $\mathcal{L}$ is the infinitesimal generator.
By Assumption~\ref{ass:diss},
$\mathbf{X}^\top \mathbf{F}_\theta(\mathbf{X}) \le -\alpha\|\mathbf{X}\|^2 + n\beta$.
Since $L$ is positive semi-definite, the Laplacian coupling satisfies
$\mathbf{X}^\top(L\otimes I_d)\mathbf{X}
= \sum_{i,j} A_{ij}\|X_i - X_j\|^2 \ge 0$,
so it only \emph{strengthens} dissipativity.
Together with Assumption~\ref{ass:nondeg},
$\operatorname{tr}(\boldsymbol{\Sigma}\boldsymbol{\Sigma}^\top) \le nd \|\sigma_\phi\|_\infty^2 < \infty$.
Substituting into~\eqref{eq:genV}:
\begin{equation}
\mathcal{L} V(\mathbf{X})
\le -2\alpha\|\mathbf{X}\|^2 + 2n\beta + nd\|\sigma_\phi\|_\infty^2
\le -\rho V(\mathbf{X}) + \kappa,
\end{equation}
for computable $\rho, \kappa > 0$.
This Foster--Lyapunov criterion~\citep[Theorem~6.1]{meyn1993stability}
implies positive Harris recurrence and geometric ergodicity.
Uniqueness follows from the non-degeneracy of $\sigma_\phi$
(Assumption~\ref{ass:nondeg}) via Doob's theorem.
The $W_2$ convergence rate is then a consequence of the log-Sobolev inequality
satisfied under~\eqref{eq:sde_matrix} with non-degenerate diffusion.
\end{proof}

\subsubsection{Euler--Maruyama Discretisation}

For discrete observation intervals $\delta t$, we simulate~\eqref{eq:sde}
via the Euler--Maruyama scheme:
\begin{equation}\label{eq:em}
X_i(t{+}\delta t)
= X_i(t)
  + \Bigl[f_\theta(X_i(t))
    + \gamma\!\sum_{j \in \mathcal{N}(i)} A_{ij}(X_j(t) - X_i(t))\Bigr]\delta t
  + \sigma_\phi(X_i(t))\sqrt{\delta t}\;\epsilon_i,
\quad \epsilon_i \sim \mathcal{N}(0, I_d).
\end{equation}
Under Assumptions~\ref{ass:lip}--\ref{ass:diss}, the Euler--Maruyama
scheme is strongly consistent with step-size error $\mathcal{O}(\delta t^{1/2})$
and inherits geometric ergodicity for sufficiently small $\delta t$
\citep{talay1990expansion}.

\subsection{Score Network and Multi-Scale Denoising Score Matching}
\label{sec:score}
% ---------------------------------------------------------

\subsubsection{Score Matching Objective}

Hyv\"{a}rinen's score matching~\citep{Hyvarinen2005score} provides a tractable
objective for learning $s_\theta(x) \approx \nabla_x \log \pi^*(x)$ without
evaluating the intractable normalising constant of $\pi^*$:
\begin{equation}\label{eq:sm}
J_{\mathrm{SM}}(\theta)
= \mathbb{E}_{\pi^*}\!\left[
    \tfrac{1}{2}\|s_\theta(x)\|^2 + \nabla_x \cdot s_\theta(x)
  \right].
\end{equation}
Minimising $J_{\mathrm{SM}}$ is equivalent to minimising the Fisher
divergence $D_F(\pi^* \| p_\theta)
= \mathbb{E}_{\pi^*}[\|s^*(x) - s_\theta(x)\|^2]$
up to a constant independent of $\theta$.
However, computing $\nabla_x \cdot s_\theta(x)$ requires the Jacobian trace,
which is $\mathcal{O}(d^2)$ in $d$.
\citet{Vincent2011dsm} showed that this cost is avoided by the equivalent
\emph{denoising score matching} (DSM) objective: for a noise-perturbed
sample $\tilde{x} = x + \sigma\epsilon$ with $\epsilon \sim \mathcal{N}(0, I_d)$,
\begin{equation}\label{eq:dsm}
J_{\mathrm{DSM}}(\theta; \sigma)
= \mathbb{E}_{x \sim \pi^*,\, \epsilon \sim \mathcal{N}(0,I_d)}\!\left[
    \left\| s_\theta(\tilde{x}, \sigma) + \frac{\epsilon}{\sigma}
    \right\|^2
  \right],
\end{equation}
which equals $J_{\mathrm{SM}}$ plus a constant.
Crucially, the target $-\epsilon/\sigma$ is available in closed form,
reducing training to a standard regression.

\subsubsection{Multi-Scale Noise Schedule}

Following \citet{Song2019ncsn}, we use $L$ logarithmically spaced noise
levels:
\begin{equation}\label{eq:schedule}
\sigma_l
= \sigma_{\min}\!\left(\frac{\sigma_{\max}}{\sigma_{\min}}\right)^{\!l/(L-1)},
\quad l = 0, 1, \ldots, L-1,
\end{equation}
with $\sigma_0 = \sigma_{\min}$ and $\sigma_{L-1} = \sigma_{\max}$.
The aggregate training loss is the noise-level-weighted sum:
\begin{equation}\label{eq:loss_score}
\mathcal{L}_{\mathrm{score}}(\theta, \phi)
= \frac{1}{L}\sum_{l=0}^{L-1}
  \lambda(\sigma_l)\;
  \mathbb{E}_{x, \epsilon}\!\left[
    \left\| s_{\theta\phi}\!\left(
        \tilde{x}_l,\, \sigma_l
      \right) + \frac{\epsilon}{\sigma_l}
    \right\|^2
  \right],
\end{equation}
where $\tilde{x}_l = x + \sigma_l\epsilon$ and
$\lambda(\sigma_l) = \sigma_l^2$ is the standard EDM weighting
\citep{Karras2022edm} that prevents large-$\sigma$ terms from dominating the
gradient.

\subsubsection{Architecture of $s_{\theta\phi}$}

The score network maps a pair $(x, \sigma) \in \mathbb{R}^d \times \mathbb{R}$
to a score estimate $s \in \mathbb{R}^d$.
\begin{enumerate}
    \item \textbf{Graph-context augmentation.}
    For node $i$, the input is augmented with the degree-normalised neighbourhood
    mean $c_i = \sum_j A_{ij} X_j / (\sum_j A_{ij} + \varepsilon)$, giving an
    effective input $z_i = [x_i^\top,\, c_i^\top]^\top \in \mathbb{R}^{2d}$.
    \item \textbf{Residual MLP.}
    $h^{(0)} = \mathrm{LayerNorm}(W_{\mathrm{in}} z_i)$,
    followed by $B = 4$ residual blocks:
    $h^{(b)} = h^{(b-1)} + W_2^{(b)}\,\mathrm{GELU}(W_1^{(b)} h^{(b-1)})$,
    $h^{(b)} \leftarrow \mathrm{LayerNorm}(h^{(b)})$.
    \item \textbf{Noise conditioning.}
    A learned embedding $e(\sigma) = W_e \log\sigma \in \mathbb{R}^H$ is
    added at the penultimate layer: $h \leftarrow h + e(\sigma)$.
    \item \textbf{EDM-scaled output.}
    The final output is $s_{\theta\phi}(x, \sigma)
    = W_{\mathrm{out}} h / \sigma$,
    which maintains the correct $\sigma^{-1}$ scaling of the score gradient
    \citep{Karras2022edm}.
\end{enumerate}

\begin{proposition}[Score estimator consistency]
\label{prop:consistency}
Let $\{x^{(k)}\}_{k=1}^N$ be i.i.d.\ samples from $\pi^*$.
As $N \to \infty$, the minimiser $\hat{s}_N$ of the empirical DSM loss
converges in $L^2(\pi^*)$ to the true score:
\begin{equation}
\mathbb{E}_{\pi^*}\!\left[\bigl\|\hat{s}_N(x, \sigma) - \nabla_x \log p_\sigma(x)\bigr\|^2\right]
\xrightarrow{N \to \infty} 0,
\end{equation}
where $p_\sigma(x) = \int \mathcal{N}(x;\, x_0, \sigma^2 I_d)\,\pi^*(x_0)\,dx_0$
is the noise-perturbed density.
Furthermore, as $\sigma_{\min} \to 0$, $\nabla_x \log p_{\sigma_{\min}}(x) \to s^*(x)$
pointwise on the support of $\pi^*$.
\end{proposition}

\begin{proof}
The first claim follows from the uniform law of large numbers applied to the
bounded DSM objective, using the Lipschitz continuity of $s_{\theta\phi}$
(Assumption~\ref{ass:lip}).
The second claim follows from the continuity of the convolution with a
Gaussian kernel: $p_\sigma \to \pi^*$ in total variation as $\sigma \to 0$
\citep[Proposition~3]{Song2019ncsn}, which implies pointwise convergence of
the score on the interior of $\mathrm{supp}(\pi^*)$.
\end{proof}

% ---------------------------------------------------------
\subsection{Joint SDE and Score Training}
\label{sec:training}
% ---------------------------------------------------------

\textsc{Siren} is trained exclusively on benign traffic.
The total loss combines two complementary objectives:
\begin{equation}\label{eq:loss_total}
\mathcal{L}_{\mathrm{total}}(\theta, \phi)
= \mathcal{L}_{\mathrm{score}}(\theta, \phi)
  + \lambda_{\mathrm{sde}}\,\mathcal{L}_{\mathrm{sde}}(\theta, \phi),
\end{equation}
where $\mathcal{L}_{\mathrm{score}}$ is defined in~\eqref{eq:loss_score} and
$\mathcal{L}_{\mathrm{sde}}$ is the one-step Euler--Maruyama prediction error:
\begin{equation}\label{eq:loss_sde}
\mathcal{L}_{\mathrm{sde}}(\theta, \phi)
= \mathbb{E}_{i, t}\!\left[
    \left\| X_i(t{+}\delta t) -
            \Bigl[X_i(t) + b_i(X(t))\,\delta t\Bigr]
    \right\|^2
  \right],
\end{equation}
where $b_i(X(t)) = f_\theta(X_i(t)) + \gamma\sum_j A_{ij}(X_j(t) - X_i(t))$
is the deterministic part of the drift.
The two losses serve complementary roles: $\mathcal{L}_{\mathrm{sde}}$
anchors the drift and diffusion parameters to observed trajectories, while
$\mathcal{L}_{\mathrm{score}}$ teaches the score network to characterise the
stationary geometry of those trajectories.

\begin{algorithm}[t]
\caption{\textsc{Siren} Training}
\label{alg:train}
\begin{algorithmic}[1]
\Require Benign flow logs $\mathcal{D}$; hyperparameters
         $\gamma, \beta, \lambda_{\mathrm{sde}}, \{\sigma_l\}_{l=0}^{L-1}$,
         $\mathrm{FAR}_{\mathrm{target}}$
\Ensure Parameters $(\theta, \phi)$; threshold $\tau^*$; normaliser $(\mu_\mathrm{tr}, \sigma_\mathrm{tr})$
\State Construct graph $\mathcal{G}$ and trajectories
       $\{X_i(t_k)\}$ from $\mathcal{D}$; fit $(\mu_\mathrm{tr}, \sigma_\mathrm{tr})$;
       split into $\mathcal{D}_\mathrm{tr}$, $\mathcal{D}_\mathrm{cal}$,
       $\mathcal{D}_\mathrm{val}$ (70/20/10 by time)
\State Initialise $s_{\theta\phi}$, $f_\theta$, $\sigma_\phi$ (Xavier);
       Adam($\eta=2\times 10^{-4}$)
\For{each mini-batch $(X_i(t), X_i(t{+}\delta t))$ from $\mathcal{D}_\mathrm{tr}$}
  \State Sample $l \sim \mathrm{Uniform}\{0,\ldots,L-1\}$,
         $\epsilon \sim \mathcal{N}(0, I_d)$
  \State $\tilde{x} \leftarrow X_i(t) + \sigma_l \epsilon$
  \State Compute $\mathcal{L}_{\mathrm{score}}$ via~\eqref{eq:loss_score}
         and $\mathcal{L}_{\mathrm{sde}}$ via~\eqref{eq:loss_sde}
  \State Update parameters via $\nabla_{\theta,\phi}\mathcal{L}_{\mathrm{total}}$;
         clip gradient norm to $1.0$
\EndFor
\State \textbf{Threshold calibration:} run inference
       (Algorithm~\ref{alg:infer}) on $\mathcal{D}_\mathrm{cal}$;
       collect $\{\tilde{R}_i(t)\}$;
       set $\tau^* = \mathrm{Quantile}\!\left(\{\tilde{R}_i\},\,
       1 - \mathrm{FAR}_{\mathrm{target}}\right)$
\end{algorithmic}
\end{algorithm}

\subsection{Inference and Anomaly Detection via Stein Residuals}
\label{sec:inference}
% ---------------------------------------------------------

\subsubsection{The Stein Operator and Identity}

For a probability measure $p$ with score $s_p(x) = \nabla_x \log p(x)$ and
a smooth vector field $g : \mathbb{R}^d \to \mathbb{R}^d$, the
\emph{Stein operator} is:
\begin{equation}\label{eq:stein_op}
\mathcal{A}_p\, g(x)
= s_p(x)^\top g(x) + \nabla_x \cdot g(x).
\end{equation}
\begin{theorem}[Graph Stein Identity]
\label{thm:stein}
Let $\pi^*$ be the unique stationary distribution of the graph-coupled
SDE~\eqref{eq:sde} (guaranteed by Theorem~\ref{thm:ergodic}).
For any smooth $g \in \mathcal{G}(\pi^*)$ in the Stein class of $\pi^*$
(i.e.\ $\pi^* g$ vanishes at infinity), and for all nodes $i \in V$:
\begin{equation}\label{eq:stein_id}
\mathbb{E}_{\pi^*}\!\left[\mathcal{A}_{\pi^*} g(X_i)\right] = 0.
\end{equation}
Consequently, the Stein residual $R_i(t) = \|s_{\theta\phi}(X_i(t))
- \hat{s}_{\mathrm{emp}}(X_i(t))\|_2$ satisfies
$\mathbb{E}_{\pi^*}[R_i(t)] = 0$ in the limit of a consistent score
estimator (Proposition~\ref{prop:consistency}).
\end{theorem}

\begin{proof}
Integration by parts on $\mathbb{R}^d$ under $\pi^*$:
\begin{align}
\mathbb{E}_{\pi^*}[\mathcal{A}_{\pi^*} g(X_i)]
&= \int \bigl[\nabla_x \log\pi^*(x)\bigr]^\top g(x)\,\pi^*(x)\,dx
   + \int \nabla_x \cdot g(x)\,\pi^*(x)\,dx \notag\\
&= \int \frac{\nabla_x \pi^*(x)}{\pi^*(x)}\cdot g(x)\,\pi^*(x)\,dx
   + \int \nabla_x \cdot g(x)\,\pi^*(x)\,dx \notag\\
&= \int \nabla_x \pi^*(x)\cdot g(x)\,dx
   + \int \nabla_x \cdot g(x)\,\pi^*(x)\,dx \notag\\
&= -\int \pi^*(x)\,\nabla_x \cdot g(x)\,dx
   + \int \nabla_x \cdot g(x)\,\pi^*(x)\,dx = 0,
\label{eq:stein_proof}
\end{align}
where the second-to-last step applies the divergence theorem using the
boundary condition $\pi^* g \to 0$ at infinity.
Under the consistent score estimator (Proposition~\ref{prop:consistency}),
$s_{\theta\phi}(x, \sigma_{\min}) \to s^*(x)$, so $R_i \to 0$ in $L^2(\pi^*)$.
\end{proof}

\subsubsection{Empirical Score Estimation and Stein Residual}

At inference, we cannot evaluate $s^*$ directly; instead we estimate it from
a sliding window $\mathcal{H}_i = \{X_i(t-k\delta t)\}_{k=0}^{W-1}$ of the
$W$ most recent observations of node $i$ using a kernel density estimator
(KDE).
For query point $x = X_i(t)$, the KDE score is:
\begin{equation}\label{eq:kde}
\hat{s}_{\mathrm{emp}}(x; \mathcal{H}_i)
= \frac{\sum_{x' \in \mathcal{H}_i} \nabla_x k_h(x, x')}
       {\sum_{x' \in \mathcal{H}_i} k_h(x, x')},
\end{equation}
where $k_h(x, x') = \exp(-\|x - x'\|^2 / 2h^2)$ is the RBF kernel with
Silverman bandwidth $h = 1.06\,\hat{\sigma}_i\, W^{-1/5}$
and $\hat{\sigma}_i$ is the empirical standard deviation of $\mathcal{H}_i$.
The gradient is $\nabla_x k_h(x, x') = -(x - x')/h^2 \cdot k_h(x, x')$.
The per-node Stein residual is then:
\begin{equation}\label{eq:stein_res}
R_i(t)
= \Bigl\| s_{\theta\phi}(X_i(t),\, \sigma_{\min})
         - \hat{s}_{\mathrm{emp}}(X_i(t);\, \mathcal{H}_i)
  \Bigr\|_2.
\end{equation}

\subsubsection{Graph-Aware Aggregation}

Individual Stein residuals may be small for individual hosts participating in
a coordinated, low-and-slow lateral-movement attack.
We therefore aggregate residuals over the graph neighbourhood:
\begin{equation}\label{eq:agg}
\tilde{R}_i(t)
= R_i(t) + \beta \sum_{j \in \mathcal{N}(i)} A_{ij}\, R_j(t),
\end{equation}
where $\beta \ge 0$ is the neighbourhood diffusion weight.
Equation~\eqref{eq:agg} can be written as $\tilde{R}(t) = (I + \beta A) R(t)$
in matrix form, i.e.\ a single step of a random walk with restart, preserving
non-negativity and amplifying residuals on dense subgraphs.

\begin{theorem}[Detection Power Bound]
\label{thm:power}
Let $P_{\mathrm{att}}$ be the distribution of traffic under an attack that
perturbs the distribution of node $i$ by KL divergence
$\delta_i = D_{\mathrm{KL}}(P_{\mathrm{att}}^{(i)} \| \pi^*)$.
Under a Lipschitz score network with constant $L_s$, the expected aggregated
Stein residual satisfies:
\begin{equation}\label{eq:power}
\mathbb{E}_{P_{\mathrm{att}}}\!\left[\tilde{R}_i(t)\right]
\ge \frac{1}{L_s}\,\mathrm{KSD}(P_{\mathrm{att}}^{(i)}, \pi^*)^2
    + \beta \sum_{j \in \mathcal{N}(i)} A_{ij}\;
      \mathbb{E}_{P_{\mathrm{att}}}\!\left[R_j(t)\right],
\end{equation}
where $\mathrm{KSD}(Q, P)^2 \ge \frac{1}{2}\bigl(1 - e^{-D_{\mathrm{KL}}(Q\|P)}\bigr)$
by Pinsker's inequality extended to the KSD.
In particular, $\mathbb{E}_{P_{\mathrm{att}}}[\tilde{R}_i(t)]$ is strictly
positive for any $\delta_i > 0$, i.e.\ \textsc{Siren} is a consistent
detector.
\end{theorem}

\begin{proof}[Proof sketch]
By the Cauchy--Schwarz inequality and the definition of the KSD:
\begin{equation}
\mathbb{E}_{P_{\mathrm{att}}}[R_i(t)]
\ge \mathbb{E}_{P_{\mathrm{att}}}\!\left[
    \bigl\langle g^*(X_i),\, s^*(X_i) - \nabla\log P_{\mathrm{att}}(X_i)
    \bigr\rangle
  \right]
= \mathrm{KSD}(P_{\mathrm{att}}^{(i)}, \pi^*)^2 / L_s,
\end{equation}
where $g^*$ is the optimal Stein witness function.
Applying the Bretagnolle--Huber--Carol inequality
$\mathrm{KSD}^2 \ge \frac{1}{2}(1 - e^{-D_{\mathrm{KL}}})$ then yields the
lower bound in terms of KL divergence.
The aggregation term $\beta \sum_j A_{ij} \mathbb{E}[R_j]$ is non-negative
by $A_{ij} \ge 0$ and non-negativity of residuals, giving the final
bound~\eqref{eq:power}.
\end{proof}

\subsubsection{Threshold Calibration and Alert Generation}

The detection threshold $\tau^*$ is set on the held-out benign calibration
set $\mathcal{D}_\mathrm{cal}$:
\begin{equation}\label{eq:threshold}
\tau^* = \mathrm{Quantile}\!\left(
    \bigl\{\tilde{R}_i(t) : (i, t) \in \mathcal{D}_\mathrm{cal}\bigr\},\;
    1 - \mathrm{FAR}_{\mathrm{target}}
\right),
\end{equation}
so that the empirical false alarm rate on $\mathcal{D}_\mathrm{cal}$ does not
exceed $\mathrm{FAR}_{\mathrm{target}} = 0.01$ by construction.
An alert is issued for node $i$ at time $t$ if $\tilde{R}_i(t) > \tau^*$.

\begin{algorithm}[t]
\caption{\textsc{Siren} Inference (per time window)}
\label{alg:infer}
\begin{algorithmic}[1]
\Require Trained $(\theta, \phi)$; threshold $\tau^*$;
         normaliser $(\mu_\mathrm{tr}, \sigma_\mathrm{tr})$;
         history $\{\mathcal{H}_i\}$;
         coupling $(\gamma, \beta)$
\Ensure Per-node alert scores $\tilde{R}_i(t)$ and flags $a_i \in \{0,1\}$
\State Parse new flows; update $A$, $L$ (decay stale edges by $0.9 \cdot A_{\mathrm{old}}$)
\State For each active node $i$: extract $X_i(t)$; normalise via
       $(\mu_\mathrm{tr}, \sigma_\mathrm{tr})$; update $\mathcal{H}_i$
\State Compute graph context $c_i \leftarrow \sum_j A_{ij} X_j /
       (\sum_j A_{ij} + \varepsilon)$ for all $i$
\State Score prediction: $s_{\mathrm{pred},i}
       \leftarrow s_{\theta\phi}\!\left([X_i^\top, c_i^\top]^\top,\,
       \sigma_{\min}\right)$ for all $i$
\State KDE score: $\hat{s}_{\mathrm{emp},i}
       \leftarrow \hat{s}_{\mathrm{emp}}(X_i; \mathcal{H}_i)$
       via~\eqref{eq:kde} (skip if $|\mathcal{H}_i| < 5$, set $R_i = 0$)
\State Stein residual: $R_i \leftarrow
       \|s_{\mathrm{pred},i} - \hat{s}_{\mathrm{emp},i}\|_2$ for all $i$
\State Aggregate: $\tilde{R}_i \leftarrow R_i + \beta \sum_{j\in\mathcal{N}(i)} A_{ij} R_j$
\State Alert: $a_i \leftarrow \mathbf{1}[\tilde{R}_i > \tau^*]$
\State \textbf{return} $\{(i, \tilde{R}_i, a_i)\}_{i \in V}$
\end{algorithmic}
\end{algorithm}

\subsection{Complexity Analysis}
\label{sec:complexity}

\textbf{Training.}
Each mini-batch step processes $B$ node-state pairs.
The dominant costs per step are the score network forward/backward pass
$\mathcal{O}(BH^2)$ and the graph-context aggregation $\mathcal{O}(B|E|)$,
where $H$ is the hidden dimension.
Total training complexity is $\mathcal{O}(N_{\mathrm{epochs}} \cdot N/B \cdot
(BH^2 + B|E|)) = \mathcal{O}(N_{\mathrm{epochs}} \cdot N(H^2 + |E|))$.

\textbf{Inference.}
For a window with $n$ active nodes and $|E|$ edges, the per-window cost is:
\begin{itemize}
    \item Score network: $\mathcal{O}(nH^2)$
    \item KDE score estimation: $\mathcal{O}(nWd)$ using the sliding window
    \item Graph aggregation: $\mathcal{O}(|E|)$ (sparse matrix-vector product)
\end{itemize}
Total inference: $\mathcal{O}(n(H^2 + Wd) + |E|)$, which is
sub-quadratic in $n$ and admits parallelisation across nodes.
For the default settings ($n \le 500$, $H = 256$, $W = 20$, $d = 12$),
the per-window latency is below 10 ms on a single CPU core.
